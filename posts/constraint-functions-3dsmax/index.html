Constraint Functions - 3DS Max

March 22, 2015 Development 3dsmax development maxscript snippets

In the past years I started creating my own library of functions and scripts that would automate a lot of the processes that I often use when rigging characters to speed up the rigging process even more.

I have so many rigging functions that one day I realized I could create an entire humanoid rig in a matter of minutes by using all these functions as long as I had a skeleton created.

In time I will be releasing a function library for all of you to check it out, meanwhile, here is a little preview of some of the functions I use often, the Look-At constraint could use more work but the Position Constraint, Orientation Constraint and Path Constraint Functions are perfect for pretty much anything that I would use them for.

  1 -- Position Constraint
  2 -- PosConst (Affected Node) (Target Node) (Target Weight)
  3 -- (newCon:true if fresh constrain, false if adding target to existing constrain)
  4 fn PosConst obj target tarWeight newCon:true =
  5 (
  6     if newCon == true then
  7     (
  8         Pc = Position_Constraint()
  9         obj.pos.controller.available.controller = Pc
 10         posConstraint = Pc.constraints
 11         posConstraint.appendtarget target tarWeight
 12     )
 13     else
 14     (
 15         obj.position.controller.Position_Constraint.appendtarget target tarWeight
 16     )
 17     obj.pos.controller.setactive 3
 18 )
 19 
 20 -- Look-At Constraint
 21 -- LookConst (Affected Node) (Target Node) (upNode:UpNode Node) (Target Weight)
 22 -- (newCon:true if fresh constrain, false if adding target to existing constrain)
 23 fn LookConst obj target upNode tarWeight newCon:true =
 24 (
 25     if newCon == true then
 26     (
 27         Lc = LookAt_Constraint()
 28         obj.rotation.controller.available.controller = Lc
 29         locConstraint = Lc.constraints
 30         locConstraint.appendtarget target tarWeight
 31         Lc.upnode_world = false
 32         Lc.pickUpNode = upNode
 33         Lc.lookat_vector_length = 0
 34     )
 35     else
 36     (
 37         obj.rotation.controller.LookAt_Constraint.appendtarget target tarWeight
 38     )
 39     obj.rotation.controller.setactive 3
 40 )
 41 
 42 -- Orientation Constraint
 43 -- OrientConst (Affected Node) (Target Node) (Target Weight) (keepOff: keep off-set True or False)
 44 -- (newCon:true if fresh constrain, false if adding target to existing constrain)
 45 fn OrientConst obj target tarWeight keepOff:false newCon:true =
 46 (
 47     if newCon == true then
 48     (
 49         Oc = Orientation_Constraint()
 50         obj.rotation.controller.available.controller = Oc
 51         orientConstraint = Oc.constraints
 52         orientConstraint.appendtarget target tarWeight
 53         Oc.relative = keepOff
 54     )
 55     else
 56     (
 57         obj.rotation.controller.Orientation_Constraint.appendtarget target tarWeight
 58     )
 59     obj.rotation.controller.setactive 3
 60 )
 61 
 62 -- Path Constraint
 63 -- PathConst (Affected Node) (Target Node) (Target weight) (Percent on path)
 64 fn PathConst obj target tarWeight percent =
 65 (
 66     Pc = Path_Constraint()
 67     obj.pos.controller.available.controller = Pc
 68     Pc.path = target
 69     deletekeys obj
 70     Pc.follow = true
 71     Pc.percent = percent
 72     obj.pos.controller.setactive 3
 73 )

 76 -- Freeze Transform -- Function created by Mathieson Facer.
 77 -- FreezeTransform (Obj to Freeze)
 78 fn FreezeTransform Obj =
 79 (
 80     local curObj = Obj
 81     if classof CurObj.rotation.controller != Rotation_Layer do
 82     (
 83         -- freeze rotation
 84         curObj.rotation.controller = Euler_Xyz()
 85         curObj.rotation.controller = Rotation_list()
 86         curObj.rotation.controller.available.controller = Euler_xyz()
 87 
 88         /* "Localization on" */
 89         CurObj.rotation.controller.setname 1 "Frozen Rotation"
 90         CurObj.rotation.controller.setname 2 "Zero Euler XYZ"
 91 
 92         /* "Localization off" */
 93         CurObj.rotation.controller.SetActive 2
 94     )
 95 
 96     if classof CurObj.position.controller != Position_Layer do
 97     (
 98         -- freeze position
 99         CurObj.position.controller = Bezier_Position()
100         CurObj.position.controller = position_list()
101         CurObj.position.controller.available.controller = Position_XYZ()
102 
103         /* "Localization on" */
104         CurObj.position.controller.setname 1 "Frozen Position"
105         CurObj.position.controller.setname 2 "Zero Pos XYZ"
106 
107         /* "Localization off" */
108         CurObj.position.controller.SetActive 2
109 
110         -- position to zero
111         CurObj.Position.controller[2].x_Position = 0
112         CurObj.Position.controller[2].y_Position = 0
113         CurObj.Position.controller[2].z_Position = 0
114     )
115 )
